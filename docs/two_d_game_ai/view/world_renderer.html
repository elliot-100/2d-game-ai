 89    def render(self, *, debug_render_mode: bool = False) -> None:
 90        """Render the `World` to `self.surface`."""
 91        self.ensure_renderers()
 92        # Drawn in order, bottom layer to top:
 93        self.surface.blit(self.base_grid_surface)
 94        self.render_untraversable_cells(self.world.grid)
 95        for b in self.bot_renderers:
 96            b.render(debug_render_mode=debug_render_mode)
 97        for m in self.obstacle_renderers:
 98            m.render()
 99        self.render_axes()

101    def base_grid(self) -> Surface:
102        """Pre-render grid to a static surface."""
103        surface = Surface((self.size, self.size))
104        surface.fill(colors.WORLD_FILL)
105
106        cell_size = self.world.grid_resolution
107        cell_offsets = [
108            cell_size * i - self.world.magnitude + cell_size / 2
109            for i in range(self.world.grid.size)
110        ]
111        for x, y in itertools.product(cell_offsets, cell_offsets):
112            self.draw_circle(
113                surface=surface,
114                color=colors.WORLD_GRID_LINE,
115                center=Vector2(x, y),
116                radius=self.GRID_NODE_RADIUS,
117                scale_radius=False,
118            )
119        return surface

121    def ensure_renderers(self) -> None:
122        """Update the set of entity renderers."""
123        for e in {e for e in self.world.entities if e.id not in self.entity_renderers}:
124            if e.id is None:
125                err_msg = f"{e!s} must have an `id` to be rendered."
126                raise ValueError(err_msg)
127
128            if is_obstacle(e):
129                self.entity_renderers[e.id] = ObstacleRenderer(parent=self, entity=e)
130            elif isinstance(e, Bot):
131                self.entity_renderers[e.id] = BotRenderer(parent=self, entity=e)
132            logger.debug(f"Added renderer for {e!s}.")

134    def to_local(self, world_pos: Vector2) -> Vector2:
135        """Convert `World` coordinates to local coordinates.
136
137        Parameters
138        ----------
139        world_pos
140            `World` coordinates
141
142        Returns
143        -------
144        Vector2
145            Local coordinates.
146            Origin is at centre, positive y upwards.
147        """
148        pos = world_pos
149        pos = pos.elementwise() * Vector2(1, -1)
150        offset = self.world.magnitude
151        pos += Vector2(offset, offset)
152        return pos * self.scale_factor

154    def to_world(self, local_pos: Vector2) -> Vector2:
155        """Convert local coordinates to `World` coordinates.
156
157        Parameters
158        ----------
159        local_pos
160            Local coordinates.
161            Origin is at centre, positive y upwards.
162
163        Returns
164        -------
165        Vector2
166            `World` coordinates.
167        """
168        pos = local_pos / self.scale_factor
169        offset = self.world.magnitude
170        pos -= Vector2(offset, offset)
171        return pos.elementwise() * Vector2(1, -1)

173    def render_untraversable_cells(self, grid: Grid) -> None:
174        """Draw the blocked cells."""
175        cell_size = self.world.grid_resolution
176        for cell_ref in grid.untraversable_cells:
177            grid_rect = Rect(
178                (cell_ref.x * cell_size, cell_ref.y * cell_size), (cell_size, cell_size)
179            )
180            # draw slightly oversize to avoid antialiasing issues
181            grid_rect = grid_rect.move(-0.5, -0.5)
182            grid_rect.width += 1
183            grid_rect.height += 1
184            self.draw_rect(color=colors.OBSTACLE_FILL, rect=Rect(grid_rect), width=0)

186    def render_axes(self) -> None:
187        """Draw axes."""
188        self.draw_line(
189            color=colors.WORLD_AXES_LINE,
190            start_pos=Vector2(0, -self.world.magnitude),
191            end_pos=Vector2(0, self.world.magnitude),
192        )
193        self.draw_line(
194            color=colors.WORLD_AXES_LINE,
195            start_pos=Vector2(-self.world.magnitude, 0),
196            end_pos=Vector2(self.world.magnitude, 0),
197        )

213    def handle_mouse_set_destination(self, local_pos: Vector2) -> Vector2 | None:
214        """Attempt to set destination, if applicable to current selection."""
215        if not isinstance(self.selected_renderer, BotRenderer):
216            return None  # Only `Bot`s support destination setting
217
218        if not isinstance(self.selected_renderer.entity, Bot):
219            raise TypeError
220
221        world_pos = self.to_world(local_pos)
222        cell = Grid.grid_ref_from_world_pos(world=self.world, pos=world_pos)
223
224        if not self.world.grid.is_traversable(cell):
225            return None
226
227        self.selected_renderer.entity.destination = world_pos
228        return world_pos

319        self,
320        *,
321        color: Color,
322        center: Vector2,
323        radius: float,
324        central_angle: float,
325        theta: float,
326    ) -> None:
327        """Draw a closed unfilled pie wedge in `World` units on `self.surface`."""
328        angle_step_degrees = 5
329        draw_angles = [
330            *list(range(0, int(theta), angle_step_degrees)),
331            theta,  # so theta is always drawn, irrespective of step
332        ]
333        offsets = [
334            Vector2(0, radius).rotate(
335                theta / 2
336                - angle
337                - central_angle,  # Pygame rotates CCW, thus negative angle
338            )
339            for angle in draw_angles
340        ]
341        self.draw_poly(
342            color=color,
343            points=[center] + [center + offset for offset in offsets],
344        )

363        blit(
364            source=source,
365            surface=self.surface,
366            dest=self.to_local(dest) + display_offset,
367        )
