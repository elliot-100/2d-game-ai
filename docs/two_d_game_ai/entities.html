 77    def update(self, other_bots: set[Bot]) -> None:
 78        """Update Bot, including move over 1 simulation step."""
 79        self._handle_sensing(other_bots)
 80
 81        if self.route and self.is_at(self.route[0]):
 82            self.notify_observers("I've reached next waypoint.")
 83            del self.route[0]
 84            self.stop()
 85
 86        if self.destination and self.is_at(self.destination):
 87            self.notify_observers("I've reached destination.")
 88            self.destination = None
 89            self.stop()
 90            return
 91
 92        if self.route:
 93            waypoint_relative_bearing = self.heading.relative(
 94                self.route[0] - self.pos
 95            ).degrees_normalised
 96
 97            #  if Bot can complete rotation to face wp this step...
 98            if abs(waypoint_relative_bearing) <= self.max_rotation_step:
 99                # face wp precisely
100                self.rotate(-waypoint_relative_bearing)
101                # initiate move towards wp
102                self.velocity = self.heading.vector * Bot.MAX_SPEED
103
104            else:
105                # turn towards wp
106                self.rotate(
107                    math.copysign(
108                        self.max_rotation_step,
109                        waypoint_relative_bearing,
110                    ),
111                )
112        self._move()

142    def _handle_sensing(self, other_bots: set[Bot]) -> None:
143        currently_visible_bots = {bot for bot in other_bots if self.can_see(bot)}
144        newly_spotted_bots = currently_visible_bots - self.visible_bots
145        newly_lost_bots = self.visible_bots - currently_visible_bots
146
147        for bot in newly_spotted_bots:
148            self.notify_observers(f"I've spotted `{bot.name}`")
149        for bot in newly_lost_bots:
150            self.notify_observers(f"I've lost sight of `{bot.name}`")
151
152        self.visible_bots.update(newly_spotted_bots)
153        self.visible_bots.difference_update(newly_lost_bots)
154        self.known_bots.update(newly_lost_bots)
155
